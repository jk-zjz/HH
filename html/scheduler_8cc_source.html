56  void Scheduler::start() {
57  MutexType::Lock lock(m_mutex);
58  if (!m_stopping) {
59  //需要启动需要判断是否是停止状态才能启动
60  return;
61  }
62  //已经启动
63  m_stopping = false;
64  //判断线程池是否为空，线程池为空则创建线程池
65  HH_ASSERT(m_threads.empty());
66  m_threads.resize(m_thread_count);
67  for (size_t i = 0; i < m_thread_count; ++i) {
68  m_threads[i].reset(new Thread(std::bind(&Scheduler::run, this),
69  m_name + "_" + std::to_string(i)));
70  m_thread_ids.push_back(m_threads[i]->get_id());
71  }
72  lock.unlock();
73  }

81  m_auto_stop = true;
82  if (m_root_fiber && m_thread_count == 0 &&
83  (m_root_fiber->getState() == Fiber::State::TERM ||
84  m_root_fiber->getState() == Fiber::State::INIT)) {
85  //主线程已经创建 && 主协程已经结束 &&（主协程是停止状态 || 主协程是初始状态）
86  m_stopping = true;

88  if (stopping()) {
89  return;
90  }
91  }
92 
93  //是否有主线程
94  if (m_root_thread != -1) {
95  HH_ASSERT(GetThis() == this)
96  } else {
97  HH_ASSERT(GetThis() != this)
98  }
99  //停止标志
100  m_stopping = true;
101  for (size_t i = 0; i < m_thread_count; ++i) {
102  //唤醒所有线程停止
103  tickle();
104  }
105  if (m_root_thread) {
106  //停止主线程
107  tickle();
108  }
112  if (m_root_fiber) {
113  if(!stopping()){
114  m_root_fiber->call();
115  }
116  }
117  std::vector<Thread::ptr> thrs;
118  {
119  MutexType::Lock lock(m_mutex);
120  thrs.swap(m_threads);
121  }
125  for (auto &i: thrs) {
126  i->join();
127  }

145  void Scheduler::run() {
146  //设置钩子函数 为不阻塞
147  set_hook_enable(true);
148  //设置为当前执行的调度器
149  setThis();
150  //判断当前线程是否为主线程
151  if (hh::GetThreadID() != m_root_thread) {
152  //设置当前线程的协程
153  t_fiber = Fiber::GetThis().get();
154  }
155  //设置空闲协程执行
156  Fiber::ptr idle_fiber(new Fiber(std::bind(&Scheduler::idle, this)));
157  //未了提供给fun函数，变成协程
158  Fiber::ptr cb_fiber;
159 
160  //取任务
161  FiberAndThread ft;
162  while (true) {
163  ft.reset();
164  //需要唤醒线程
165  bool tickle_me = false;
166  bool is_active = false;
167  {
168  MutexType::Lock lock(m_mutex);
169  auto it = m_fibers.begin();
170  while (it != m_fibers.end()) {
171  if (it->thread_id != (uint32_t) -1 && it->thread_id != (uint32_t) hh::GetThreadID()) {
172  tickle_me = true;
173  it++;
174  continue;
175  }
176 
177  HH_ASSERT(it->fiber || it->function);
178  if (it->fiber && it->fiber->getState() == Fiber::EXEC) {
179  it++;
180  continue;
181  }
182  ft = *it;
183  m_fibers.erase(it++);
184  ++m_active_thread_count;
185  is_active = true;
186  break;
187  }
188  tickle_me |= it != m_fibers.end();
189  }
190  //任务已经取出获取没有任务
191  if (tickle_me) {
192  //指定了执行协程，去重新唤醒
193  tickle();
194  }
195  //是协程任务，并且不是停止态
196  if (ft.fiber && (ft.fiber->getState() != Fiber::State::TERM &&
197  ft.fiber->getState() != Fiber::EXCEPT)) {
198 
199  //切换执行
200  ft.fiber->swapIn();
201  --m_active_thread_count;
202  if (ft.fiber->getState() == Fiber::State::READY) {
203  //这一个加入队列
204  schedule(ft.fiber);
205  } else if (ft.fiber->getState() != Fiber::TERM &&
206  ft.fiber->getState() != Fiber::State::EXCEPT) {
207  //不是停止态或者异常 被打断
208  ft.fiber->setState(Fiber::State::HOLD);
209  }
210  ft.reset();
211  } else if (ft.function) {
212  if(cb_fiber) {
213  cb_fiber->reset(ft.function);
214  } else {
215  cb_fiber.reset(new Fiber(ft.function));
216  }
217  ft.reset();
218  cb_fiber->swapIn();
219  --m_active_thread_count;
220  if(cb_fiber->getState() == Fiber::READY) {
221  schedule(cb_fiber);
222  cb_fiber.reset();
223  } else if(cb_fiber->getState() == Fiber::EXCEPT
224  || cb_fiber->getState() == Fiber::TERM) {
225  cb_fiber->reset(nullptr);
226  } else {//if(cb_fiber->getState() != Fiber::TERM) {
227  cb_fiber->setState(Fiber::State::HOLD);
228  cb_fiber.reset();
229  }
230  } else {
231  if (is_active) {
232  --m_active_thread_count;
233  continue;
234  }
235  //是一个空闲协程
236  if (idle_fiber->getState() == Fiber::State::TERM) {
237  break;
238  }
239  ++m_idle_thread_count;
240  idle_fiber->swapIn();
241  --m_idle_thread_count;
242  if (idle_fiber->getState() != Fiber::State::TERM
243  && idle_fiber->getState() != Fiber::State::EXCEPT) {
244  idle_fiber->setState(Fiber::State::HOLD);
245  }
246  }
247  }
