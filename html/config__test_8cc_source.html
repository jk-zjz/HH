9 void printNode(const YAML::Node& node) {
10  switch (node.Type()) {
11  case YAML::NodeType::Null:
12  std::cout << "null" << std::endl;
13  break;
14  case YAML::NodeType::Scalar:
15  std::cout <<" " <<node.as<std::string>() << std::endl;
16  break;
17  case YAML::NodeType::Sequence:
18  for (size_t i = 0; i < node.size(); ++i) {
19 
20  printNode(node[i]);
21  }
22  break;
23  case YAML::NodeType::Map:
24  for (YAML::const_iterator it = node.begin(); it != node.end(); ++it) {
25  std::cout <<"key "<< it->first.as<std::string>() << "\n";
26  printNode(it->second);
27  }
28  break;
29  default:
30  std::cout << "unknown type" << std::endl;
31  break;
32  }
33 }
