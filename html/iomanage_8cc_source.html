77  int IOManager::addEvent(int fd, IOManager::Event event, std::function<void()> cb) {
78  FdContext *FdCtx = nullptr;
79  //读锁判断
80  RWMutexType::ReadLock lock(m_RWMutex);
81  if ((int) m_fdContexts.size() > fd) {
82  FdCtx = m_fdContexts[fd];
83  lock.unlock();
84  } else {
85  lock.unlock();
86  //写锁扩容
87  RWMutexType::WriteLock lock2(m_RWMutex);
88  fdcontextReset(fd * 1.5);
89  FdCtx = m_fdContexts[fd];
90  }
91  HH_ASSERT(FdCtx);
92  //互斥锁操作
93  FdContext::MutexType::Lock lock3(FdCtx->mutex);
98  if (FdCtx->events & event) {
99  HH_LOG_LEVEL_CHAIN(g_logger, LogLevel::ERROR) << "addEvent assert fd =" << fd
100  << " event =" << event << " fdCtx.events =" << FdCtx->events;
101  HH_ASSERT(!(FdCtx->events & event));
102  }
103  //如果无事件 0 就条件事件EPOLL_CTL_ADD 否则 就修改事件EPOLL_CTL_MOD
104  int op = FdCtx->events ? EPOLL_CTL_MOD : EPOLL_CTL_ADD;
105  epoll_event ev;
106  memset(&ev, 0, sizeof(ev));
107  ev.events = EPOLLET | FdCtx->events | event;
108  ev.data.ptr = FdCtx;
109  int rt = epoll_ctl(m_EpollFd, op, fd, &ev);
110  if (rt) {
111  HH_LOG_LEVEL_CHAIN(g_logger, LogLevel::ERROR)
112  << "epoll_ctl assert fd =" << fd
113  << " event =" << event << " fdCtx.events =" << FdCtx->events
114  << " rt =" << rt << " errno =" << errno << " errstr =" << strerror(errno);
115  return -1;
116  }
117  m_pendingEventCount++;
118  FdCtx->events = (Event) (FdCtx->events | event);
119  //回调的上下文 读 | 写
120  FdContext::EventContext &event_ctx = FdCtx->getEventContext(event);
121  HH_ASSERT(!event_ctx.scheduler
122  && !event_ctx.cb
123  && !event_ctx.fiber);
124  event_ctx.scheduler = Scheduler::GetThis();
125  //指定回调函数
126  if (cb) {
127  event_ctx.cb.swap(cb);
128  } else {
129  //为指定回调NULL 添加主协程并且为执行态
130  event_ctx.fiber = Fiber::GetThis();
131  HH_ASSERT(event_ctx.fiber->getState() == Fiber::State::EXEC);
132  }
133  return 0;
134  }

142  bool IOManager::delEvent(int fd, IOManager::Event event) {
143  FdContext *FdCtx = nullptr;
144  RWMutexType::ReadLock lock(m_RWMutex);
145  if (fd >= (int) m_fdContexts.size()) {
146  return false;
147  }
148  FdCtx = m_fdContexts[fd];
149  lock.unlock();
150  FdContext::MutexType::Lock lock2(FdCtx->mutex);
151  //取出的句柄状态是否和事件一致
152  if (!(FdCtx->events & event)) {
153  //不一致
154  return false;
155  }
156  //删除事件
157  Event mew_event = (Event) (FdCtx->events & ~event);
158  int op = mew_event ? EPOLL_CTL_MOD : EPOLL_CTL_DEL;
159  epoll_event ev;
160  memset(&ev, 0, sizeof(ev));
161  ev.events = EPOLLET | mew_event;
162  ev.data.ptr = FdCtx;
163  int rt = epoll_ctl(m_EpollFd, op, fd, &ev);
164  if (rt) {
165  HH_LOG_LEVEL_CHAIN(g_logger, LogLevel::ERROR)
166  << "epoll_ctl assert fd =" << fd
167  << " event =" << mew_event << " fdCtx.events =" << FdCtx->events
168  << " rt =" << rt << " errno =" << errno << " errstr =" << strerror(errno);
169  return false;
170  }
171  --m_pendingEventCount;
172  FdCtx->events = mew_event;
173  FdContext::EventContext& event_ctx = FdCtx->getEventContext(event);
174  FdCtx->resetEventContext(event_ctx);
175  return true;
176  }

329  void IOManager::idle() {
330  //创建事件触发数组
331  const int m_epollEventSize = 255;
332  epoll_event *events = new epoll_event[m_epollEventSize]();
333  std::shared_ptr<epoll_event> event_ptr(events, [](epoll_event *ptr) {
334  delete[] ptr;
335  });
336  while (true) {
337  //引入定时器
338  uint64_t next_timeout = 0;
339  if (HH_UNLIKELY(stopping(next_timeout))) {
340  HH_LOG_LEVEL_CHAIN(g_logger, LogLevel::INFO) << "idle stopping";
341  break;
342  }
343  int rt = 0;
344  do {
345  static const int MAX_TIMEOUT = 3000;
346  if(next_timeout != ~0ull) {
347  next_timeout = (int)next_timeout > MAX_TIMEOUT
348  ? MAX_TIMEOUT : next_timeout;
349  } else {
350  next_timeout = MAX_TIMEOUT;
351  }
352  rt = epoll_wait(m_EpollFd, events, m_epollEventSize, (int)next_timeout);
353  if(rt < 0 && errno == EINTR) {
354  } else {
355  break;
356  }
357  } while(true);
358  //获取释放有需要执行的过期定时器任务
359  std::vector<std::function<void()>> cbs;
360  listExpiredCb(cbs);
361  if (!cbs.empty()) {
362  //添加到调度器中
363  schedule(cbs.begin(), cbs.end());
364  cbs.clear();
365  }
366 
367  //触发了
368  for (int i = 0; i < rt; ++i) {
369  //取出一个触发事件
370  epoll_event &event = events[i];
371  if (event.data.fd == m_TickleFd[0]) {
372  uint8_t dummy[256];
373  //读出数据，清空管到 因为使用到边沿不然就触发不了了
374  while (read(m_TickleFd[0], dummy, sizeof(dummy)) > 0);
375  continue;
376  }
377  //取出上下文
378  FdContext *FdCtx = (FdContext *) event.data.ptr;
379  FdContext::MutexType::Lock lock(FdCtx->mutex);
380  //判断event.events中是否有错误事件
381  if (event.events & (EPOLLERR | EPOLLHUP)) {
382  //没有就为event.events设置读写
383  event.events |= (EPOLLIN | EPOLLOUT) & FdCtx->events;
384  }
385  int real_events = NONE;
386  if (event.events & EPOLLIN) {
387  //触发读事件
388  real_events |= Event::READ;
389  }
390  if (event.events & EPOLLOUT) {
391  //触发写事件
392  real_events |= Event::WRITE;
393  }
394  if ((FdCtx->events & real_events) == NONE) {
395  continue;
396  }
397  // -去掉触发事件 剩余的事件
398  int left_events = (FdCtx->events & ~real_events);
399  int op = left_events ? EPOLL_CTL_MOD : EPOLL_CTL_DEL;
400  event.events = EPOLLET | left_events;
401  int rt2 = epoll_ctl(m_EpollFd, op, FdCtx->fd, &event);
402  if (rt2) {
403  HH_LOG_LEVEL_CHAIN(g_logger, LogLevel::ERROR)
404  << "epoll_ctl assert fd =" << FdCtx->fd
405  << " event =" << FdCtx->events << " fdCtx.events =" << FdCtx->events
406  << " rt =" << rt2 << " errno =" << errno << " errstr =" << strerror(errno);
407  continue;
408  }
409  //触发事件
410  if (real_events & Event::READ) {
411  FdCtx->triggerEvent(Event::READ);
412  --m_pendingEventCount;
413  }
414  if (real_events & Event::WRITE) {
415  FdCtx->triggerEvent(Event::WRITE);
416  --m_pendingEventCount;
417  }
418  }
419 
420  Fiber::ptr cur = Fiber::GetThis();
421  auto it = cur.get();
422  cur.reset();
423  it->swapOut();
424  }
425  }
