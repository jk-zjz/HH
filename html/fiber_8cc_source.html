63  Fiber::Fiber(std::function<void()> cb, size_t stackSize, bool useCaller) :
64  m_cb(std::move(cb)),
65  m_id(++s_fiber_id) {
66  ++s_fiber_count; // 增加协程计数器，用于统计当前系统中的协程数量
67  // 设置堆栈大小，如果传入的stackSize为0，则使用默认堆栈大小
68  m_stackSize = stackSize ? stackSize : g_fiber_stack_size->getValue();
69  // 为协程分配堆栈内存
70  m_stack = StackAllocator::Alloc(m_stackSize);
71  // 初始化协程的上下文结构体
72  if (getcontext(&m_ctx)) {
73  HH_ASSERT2(false, "getcontext"); // 如果getcontext失败，断言失败
74  }
75  // 配置上下文结构体的链接和堆栈信息
76  m_ctx.uc_link = nullptr;
77  m_ctx.uc_stack.ss_sp = m_stack;
78  m_ctx.uc_stack.ss_size = m_stackSize;
79  // 根据useCaller参数决定使用哪种方式初始化协程上下文
80  if (!useCaller) {
81  // 正常协程执行的方法
82  makecontext(&m_ctx, (void (*)()) &Fiber::MainFunc, 0);
83  } else {
84  // 在调度器中使用了一个线程当调度线程的协程的方法
85  makecontext(&m_ctx, (void (*)()) &Fiber::CallerMainFunc, 0);
86  }
87  }

247  void Fiber::MainFunc() {
248  Fiber::ptr m_fiber = GetThis(); // 获取当前Fiber对象
249  HH_ASSERT(m_fiber); // 检查Fiber对象是否有效
250  try {
251  m_fiber->m_cb(); // 执行Fiber回调
252  m_fiber->m_cb = nullptr; // 清空回调
253  m_fiber->m_state = TERM; // 设置状态为终止
254  } catch (std::exception &e) {
255  m_fiber->m_state = EXCEPT; // 设置状态为异常
256  // 记录异常日志
257  HH_LOG_LEVEL_CHAIN(g_logger,hh::LogLevel::ERROR)<<"Fiber Except: "<<e.what();
258  } catch (...) {
259  m_fiber->m_state = EXCEPT; // 设置状态为异常
260  // 记录未知异常日志
261  HH_LOG_LEVEL_CHAIN(g_logger,hh::LogLevel::ERROR)<<"Fiber Except: "<<"unknown exception";
262  }
263  auto pFiber = m_fiber.get(); // 获取Fiber指针
264  m_fiber.reset(); // 释放Fiber对象
265  pFiber->swapOut(); // 切换到调度器
266  HH_ASSERT2(false, "never reach fiber::mainfunc"); // 断言不应到达此位置
267  }
