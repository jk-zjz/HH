150  class CaseInsensitiveLess {
151  public:
152  bool operator()(const std::string &lhs, const std::string &rhs) const {
153  return strcasecmp(lhs.c_str(), rhs.c_str())<0;
154  }
155  };
165  template<class MapType ,class T>
166  bool checkGetAs(const MapType &m, const std::string &key, T &value, const T &def = T()) {
167  auto it = m.find(key);
168  if (it == m.end()) {
169  value = def;
170  return false;
171  }
172  try {
173  value = boost::lexical_cast<T>(it->second);
174  return true;
175  } catch (...) {
176  value = def;
177  }
178  return false;
179  }
180 
189  template<class MapType ,class T>
190  T getAs(const MapType &m, const std::string &key, const T &def = T()) {
191  auto i = m.find(key);
192  if (i == m.end()) {
193  return def;
194  }
195  try {
196  return boost::lexical_cast<T>(i->second);
197  } catch (...) {
198  return def;
199  }
200  return def;
201  }
