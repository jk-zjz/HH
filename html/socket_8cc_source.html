99  bool Socket::bind(const Address::ptr &address) {
100  if(!isValid()){
101  // socket 无效
102  // 创建socket
103  newSock();
104  if(HH_UNLIKELY(!isValid())){
105  // 还是无效
106  HH_LOG_LEVEL_CHAIN(g_logger,hh::LogLevel::DEBUG)<<"newSock error,errno="
107  <<errno<<",error="<<strerror(errno);
108  close();
109  return false;
110  }
111  }
112  // 存储的和传入地址族不匹配
113  if(HH_UNLIKELY(address->getFamily() != m_family)){
114  HH_LOG_LEVEL_CHAIN(g_logger,hh::LogLevel::DEBUG)<<"address family error"
115  <<address->getFamily()
116  <<" "<<m_family
117  <<address->toString();
118  close();
119  return false;
120  }
121  // 绑定失败不关闭文件描述符
122  if(::bind(m_sock, address->getAddr(), address->getAddrLen())){
123  HH_LOG_LEVEL_CHAIN(g_logger,hh::LogLevel::DEBUG)<<"bind error,errno="
124  <<errno<<",error="<<strerror(errno);
125  return false;
126  };
127  getLocalAddress();
128  return true;
129  }

165  close();

170  getRemoteAddress();

283  Address::ptr Socket::getLocalAddress() {
284  if(m_localAddress){
285  return m_localAddress;
286  }
287  Address::ptr ret;
288  switch(m_family){
289  case AF_INET:
290  ret.reset(new IPv4Address());
291  break;
292  case AF_INET6:
293  ret.reset(new IPv6Address());
294  break;
295  case AF_UNIX:
296  ret.reset(new UnixAddress());
297  default:
298  ret.reset(new UnknownAddress(m_family));
299  break;
300  }
301  socklen_t len = ret->getAddrLen();
302  if(getsockname(m_sock, ret->getAddr(), &len)==-1){
303  HH_LOG_LEVEL_CHAIN(g_logger,hh::LogLevel::DEBUG)<<"getLocalAddress error,errno="
304  <<errno<<",error="<<strerror(errno);
305  return Address::ptr(new UnknownAddress(m_family));
306  }
307  if(m_family==AF_UNIX){
308  UnixAddress::ptr address = std::dynamic_pointer_cast<UnixAddress>(ret);
309  if(address){
310  address->setAddrLen(len);
311  }
312  }
313  m_localAddress=ret;
314  return m_localAddress;
315  }
