248 // %s %d{yyyy}
249 //m_pattern %C %d{YYYY-MM-dd hh:mm:ss} [%m] %n
253  void LogFormatter::init() {
254  std::vector<std::tuple<std::string, std::string, int>> vec;
255  std::string nstr;
256  bool type = false;
257  for (size_t i = 0; i < m_pattern.size(); i++) {
258  if (!type && isspace(m_pattern[i])) {
259  continue;
260  }
261  while (i < m_pattern.size() && m_pattern[i] != '%') {
262  type = true;
263  nstr.append(1, m_pattern[i]);
264  i++;
265  }
266  if (!nstr.empty()) {
267  if (nstr[nstr.size() - 1] == ' ') {
268  nstr.erase(nstr.size() - 1);
269  }
270  vec.push_back(std::make_tuple(nstr, "", 0));
271  nstr.clear();
272  type = false;
273  i--;
274  continue;
275  }
276  std::string back_S;
277  std::string end_S;
278 
279  int n = i + 1;
280  if (static_cast<std::size_t>(n + 1) < m_pattern.size() && m_pattern[n] == '%' && m_pattern[n + 1] == '%') {
281  // %%
282  vec.push_back(std::make_tuple("%", std::string(), 0));
283  } else {
284  if (static_cast<std::size_t>(n) < m_pattern.size()) {
285  back_S = m_pattern.substr(n, 1);
286  if (m_pattern[n + 1] == '{') {
287  n++;
288  while (static_cast<std::size_t>(n) < m_pattern.size() && m_pattern[n] != '}') {
289  if (m_pattern[n] == '%' && m_pattern[i + 1] != 'd') {
290  n--;
291  break;
292  }
293  n++;
294  }
295  if (m_pattern[n] == '}') {
296  end_S = m_pattern.substr(i + 3, n - i - 3);
297  vec.push_back(std::make_tuple(back_S, end_S, 2));
298  } else {
299  vec.push_back(std::make_tuple("<<error>>", std::string(), 0));
300  m_Error = true;
301  }
302  } else {
303  vec.push_back(std::make_tuple(back_S, "", 1));
304  }
305  } else {
306  vec.push_back(std::make_tuple(back_S, "", 1));
307  }
308  i = n;
309  }
310  }

312  * m 日志器名称
313  * p 日志级别
314  * r 启动时间
315  * t 线程号
316  * F 协程号
317  * n 回车

323  * N 线程name

325 //是一个lambda表达式通过传入的string 返回一个格式器类每一种格式对应一个输出方式
326  static std::map<std::string, std::function<LogFormatter::FormatItem::ptr(
327  const std::string &str)> > s_format_times =
328  {
329 #define XX(str, C) \
330  {#str,[](const std::string& fmt){return LogFormatter::FormatItem::ptr(new C(fmt));}}
331  XX(m, NameFormatItem),
332  XX(p, LevelFormatItem),
333  XX(r, ElapseFormatItem),
334  XX(t, ThreadFormatItem),
335  XX(F, FiberFormatItem),
336  XX(n, EndlFormatItem),
337  XX(d, DateTimeFormatItem),
338  XX(f, FileNameFormatItem),
339  XX(l, LineFormatItem),
340  XX(b, SpaceFormatItem),
341  XX(C, MassageFormatItem),
342  XX(N, StringThreadNameItem)
343 #undef XX
344  };
345  for (auto &i: vec) {
346  if (std::get<2>(i) == 0) {
347  //字符串
348  m_items.push_back(LogFormatter::FormatItem::ptr(new StringFormatItem(std::get<0>(i))));
349  } else {
350  auto it = s_format_times.find(std::get<0>(i));
351  if (it == s_format_times.end()) {
352  //error
353  m_items.push_back(LogFormatter::FormatItem::ptr(
354  new StringFormatItem("<<error_format %" + std::get<0>(i) + ">>")));
355  m_Error = true;
356  } else {
357  m_items.push_back(it->second(std::get<1>(i)));
358  }
359  }
360  }
361  m_items.push_back(FormatItem::ptr(new MassageFormatItem()));
362  }
