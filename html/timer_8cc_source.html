175  void TimerManager::listExpiredCb(std::vector<std::function<void()>> &cbs) {
176  uint64_t now_ms = hh::GetCurrentMS();
177  std::vector<Timer::ptr> tmp;
178  {
179  RWMutexType::ReadLock lock(m_mutex);
180  if (m_timers.empty()) {
181  return;
182  }
183  }
184  RWMutexType::WriteLock lock(m_mutex);
185  if(m_timers.empty()){
186  return;
187  }
188  //是否回拨
189  bool rollover = detectClockRollover(now_ms);
190  if(!rollover && now_ms<(*m_timers.begin())->m_next_time){
191  //没有回拨 && 第一个定时器时间大于当前时间 没有过期任务
192  return;
193  }
194  Timer::ptr now_timer(new Timer(now_ms));
195  // 回拨是一定不会有超时
196  auto it =rollover ?m_timers.end() : m_timers.upper_bound(now_timer);
197 
198  while (it != m_timers.end() && (*it)->m_next_time == now_ms) {
199  it++;
200  }
201  tmp.insert(tmp.begin(), m_timers.begin(), it);
202  m_timers.erase(m_timers.begin(), it);
203  cbs.resize(tmp.size());
204  for(auto& timer : tmp) {
205  cbs.push_back(timer->m_cb);
206  if(timer->m_repeat) {
207  timer->m_next_time = now_ms + timer->m_interval;
208  m_timers.insert(timer);
209  } else {
210  timer->m_cb = nullptr;
211  }
212  }
213  }
