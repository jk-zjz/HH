137  hints.ai_protocol = protocol;

139  std::string node;
140  const char *service = nullptr;
141 
142  if (!host.empty() && host[0] == '[') {
143  // 是否是IPv6地址

145  if (endipv6) {
146  if (*(endipv6 + 1) == ':') {
147  service = endipv6 + 2;
148  }
149  node = host.substr(1, endipv6 - host.c_str() - 1);
150  }
151  } else {
152  service = (const char *)memchr(host.c_str(), ':', host.size());
153  if (service) {
154  node = host.substr(0, service - host.c_str());
155  service++;
156  }
157  }
158 
159  if (node.empty()) {
160  node = host;
161  }
162 
163  HH_LOG_LEVEL_CHAIN(g_logger, hh::LogLevel::INFO) << "Resolving node: " << node << ", service: " << (service ? service : "null");
164 
165  int error = getaddrinfo(node.c_str(), service, &hints, &results);
166  if (error) {
167  std::cerr << "Address::lookup(" << host << ", " << family << ", " << type << ", " << protocol
168  << ") error=" << error << " errstr=" << gai_strerror(error) << std::endl;
169  return false;
170  }
171 
172  next = results;
173  while (next) {
174  result.push_back(create(next->ai_addr, (socklen_t)next->ai_addrlen));

231  Address::GetInterfaceAddresses(std::multimap<std::string, std::pair<Address::ptr, uint32_t>> &result, int family) {
232  struct ifaddrs *next, *results;
233  // 获取接口地址列表
234  if(getifaddrs(&results) != 0){
235  HH_LOG_LEVEL_CHAIN(g_logger,hh::LogLevel::DEBUG) << "Address::GetInterfaceAddresses failed, errno="
236  << errno << " errstr=" << strerror(errno);
237  return false;
238  }
239  try {
240  // 循环遍历接口地址列表
241  for(next = results; next; next = next->ifa_next){
242  Address::ptr address;
243  uint32_t prefix_len = 0;
244  if(family != AF_UNSPEC && next->ifa_addr->sa_family != family){
245  continue;
246  }
247  switch (next->ifa_addr->sa_family) {
248  case AF_INET: {
249  address = create(next->ifa_addr, sizeof(sockaddr_in));
250  uint32_t netmask = ((sockaddr_in *)next->ifa_netmask)->sin_addr.s_addr;
251  prefix_len = countBits(netmask);
252  break;
253  }
254  case AF_INET6: {
255  HH_LOG_INFO(g_logger, "Address::GetInterfaceAddresses family=AF_INET6");
256  address = create(next->ifa_addr, sizeof(sockaddr_in6));
257  in6_addr& netmask_ = ((sockaddr_in6 *)next->ifa_netmask)->sin6_addr;
258  prefix_len = 0;
259  for (int i = 0; i < 16; ++i) {
260  prefix_len += countBits(netmask_.s6_addr[i]);
261  }
262  break;
263  }
264  default:
265  break;
266  }
267  if(address){
268  result.insert(std::make_pair(next->ifa_name, std::make_pair(address, prefix_len)));
269  }
270  }
271  }catch(...){
272  HH_LOG_LEVEL_CHAIN(g_logger,hh::LogLevel::DEBUG)<< "Address::GetInterfaceAddresses except";
273  freeifaddrs(results);
274  return false;
275  }
276  freeifaddrs(results);
277  return true;
278  }
