548  }

563  m_position += size;
564  size = 0;
565  bpos += size;
566  } else {
567  // 如果大于剩余容量

569  m_position += ncap;
570  size -= ncap;

614  bool ByteArray::writeToFile(const std::string &name) const {
615  std::ofstream ofs;
616  ofs.open(name, std::ios::trunc | std::ios::binary);
617  if (!ofs) {
618  HH_LOG_LEVEL_CHAIN(g_logger, hh::LogLevel::DEBUG) << "open file " << name << " failed" <<
619  "error " << errno << " " << strerror(errno);
620  return false;
621  }
622 
623  // 获取剩余大小
624  int64_t read_size = getRemainSize();
625  // 获取偏移量
626  int64_t pos = m_position;
627  // 现在在使用的页
628  Node *cur = m_cur;
629 
630  while (read_size > 0) {
631  int diff = pos % m_baseSize;
632  int64_t len = (read_size > (int64_t) m_baseSize ? m_baseSize : read_size) - diff;
633  ofs.write(cur->ptr + diff, len);
634  read_size -= len;
635  pos += len;
636  cur = cur->next;
637  }
638  return true;

680  void ByteArray::addCapacity(size_t size) {
681  if (size == 0) {
682  return;
683  }
684  // 获取剩余容量
685  size_t old_capacity = getCapacity();
686  if (old_capacity >= size) {
687  // 修改的空间没有原来大
688  return;
689  }
690  // > 多少容量
691  size = size - old_capacity;
692  // 计算需要扩容多少页
693  int cont = size / m_baseSize + ((size_t) (size % m_baseSize) > old_capacity ? 1 : 0);
694  // 找到最后一个节点
695  Node *tmp = m_root;
696  while (tmp->next) {
697  tmp = tmp->next;
698  }
699 
700  Node *first = nullptr;
701  for (int i = 0; i < cont; i++) {
702  tmp->next = new Node(m_baseSize);
703  if (first == nullptr) {
704  first = tmp->next;
705  }
706  tmp = tmp->next;
707  m_capacity += m_baseSize;
708  }
709  if (old_capacity == 0) {
710  // 剩余容量==0 代表需要到下一个节点
711  m_cur = first;
712  }
713  }
